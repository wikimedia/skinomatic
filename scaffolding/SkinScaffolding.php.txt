/**
 * Skin subclass for Vector
 * @ingroup Skins
 */
class Skin{{skin}} extends SkinTemplate {
	public $skinname = '{{skinkey}}';
	public $stylename = '{{skinkey}}';
	public $template = '{{skin}}Template';

	/**
	 * @inheritDoc
	 * @return array
	 */
	public function getDefaultModules() {
		$modules = parent::getDefaultModules();
		// add vector skin styles and vector module
		$modules['styles']['skin'][] = 'skins.{{skinkey}}.styles';

		return $modules;
	}

	/**
	 * Set up
	 *
	 * @param string $classname
	 * @return {{skin}}Template
	 */
	public function setupTemplate( $classname ) {
		$template = new {{skin}}Template( $this->getConfig() );
		$template->setTemplateParser( new TemplateParser( __DIR__ . '/src' ) );
		return $template;
	}
}
